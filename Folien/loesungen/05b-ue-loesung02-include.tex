\begin{itemize}
	\item Flexionsendungen, die historisch zugrunde gelegen haben könnten:
	
	\begin{itemize}
		\settowidth\jamwidth{[vorangestellte Genitivattribute]}
		
		\item Herz\alertred{ens}$+$angelegenheit, Land\alertred{es}$+$ministerium
		\jambox{[vorangestellte Genitivattribute]}

		\item Häus\alertred{er}$+$front, Staat\alertred{en}$+$gemeinschaft
		\jambox{[Plural]}
	\end{itemize}

\pause 
	
	\item Probleme gibt es jedoch \dots
	
	\settowidth\jamwidth{[keine Genitivalternation bei Fuge]} 
	
		\dots\ formal:
		
		\begin{itemize}
		\item Lieb\alertred{es}brief \jambox{[formal falscher Genitiv]}
		
		\end{itemize}
		\dots\ semantisch:
		
		\begin{itemize}
		\item Liebling\alertred{s}getränk \jambox{[semantisch falscher Genitiv]}
		
		\item Hühn\alertred{er}ei, Scheib\alertred{en}wischer \jambox{[semantisch falscher Plural]}
	
		\item Freund\alertred{es}kreis, Bischof\alertred{s}konferenz \jambox{[semantisch falscher Singular]}
		
		\end{itemize}
		\dots\ bei alternierenden Formen:
		
		\begin{itemize}
		\item Ende des Jahr\alertblue{es}/Jahr\alertblue{s} \jambox{[Genitivalternation]}	
		
			\vs Jahr\alertred{es}zahl/*Jahr\alertred{s}zahl \jambox{[keine Genitivalternation bei Fuge]}
\end{itemize}
\end{itemize}

\end{frame}
